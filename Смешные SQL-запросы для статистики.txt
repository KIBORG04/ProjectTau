Выбор средней длины флавора в зависимости от расы и пола.
select species, gender, avg(len) from (select gender, species, char_length(flavor) as len
                         from manifest_entries
                         where flavor <> '' AND char_length(flavor) > 10 AND name NOT LIKE 'syndicate drone%' AND flavor in (select distinct flavor from manifest_entries)) as gl
group by gender, species
order by species;
